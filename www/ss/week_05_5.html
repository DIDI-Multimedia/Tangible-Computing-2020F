Arrays

Review of Arrays



    // arrays are used to store multiple values 
    var cars = ["Saab", "Volvo", "BMW"];

    // retrieving a value from an array using an index 

    console.log(cars[0]) // prints 'Saab' to console 


    // changing an array element
    cars[0] = "Lamborghini"
    console.log(cars[0]) // prints 'Lamborghini' to console 

    // arrays are a special type of javascript object 
    var personArr = ["John", "Doe", 46];
    // or
    var personObj = {firstName:"John", lastName:"Doe", age:46};

    console.log(personArr[0]) // "John"

    // objects use names to access its properties rather than indices 

    console.log(personObj.firstName) // "John"
    console.log(personObj['firstName']) // "John"

What is "THIS"



    var person = {
      firstName: "John",
      lastName : "Doe",
      id       : 5566,
      fullName : function() {
        return this.firstName + " " + this.lastName;
      }
    };

    person.fullName() // prints 'John Doe'
